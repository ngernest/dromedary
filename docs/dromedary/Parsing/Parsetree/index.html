Module Parsing.Parsetree

Parsetree is the abstract syntax tree produced by parsing Dromedary's source code.

type core_type =
  1. | Ptyp_var of string
    (*

    Type variables 'a.

    *)
  2. | Ptyp_arrow of core_type * core_type
    (*

    Function types T1 -> T2.

    *)
  3. | Ptyp_tuple of core_type list
    (*

    Product (or "tuple") types.

    *)
  4. | Ptyp_constr of core_type list * string
    (*

    A type constructor (or "type former"), name "constr" is used for consistency between OCaml and Dromedary code.

    *)
  5. | Ptyp_variant of row
    (*

    Polymorphic variants [ ... ]

    *)
  6. | Ptyp_mu of string * core_type
    (*

    Equi-recursive types mu 'a. T

    *)
  7. | Ptyp_where of core_type * string * core_type
    (*

    where-binding T where 'a = T

    *)
and row = row_field list * closed_flag
and row_field =
  1. | Row_tag of string * core_type option
    (*

    `A `A of T

    *)
and closed_flag =
  1. | Closed
    (*

    (< | =) row

    *)
  2. | Open
    (*

    > row

    *)

type core_scheme = string list * core_type

core_schemes are defined by the grammar: core_schemes ::= 'a1 ... 'an. T

This "source"-level notion of type schemes is often used in annotations in expression and pattern.

type expression =
  1. | Pexp_var of string
    (*

    x.

    *)
  2. | Pexp_prim of Ast_types.primitive
    (*

    Primitive operations %prim p. e.g. %prim +, %prim -, etc.

    *)
  3. | Pexp_const of Ast_types.constant
    (*

    Constants c. e.g. 1, true, ().

    *)
  4. | Pexp_fun of pattern * expression
    (*

    The function (or lambda) abstraction fun P -> E. Note that:

    • let x P1 ... Pn = E in ... is encoding using Pexp_let ("x", fun P1 ... Pn -> E, ...).
    *)
  5. | Pexp_app of expression * expression
    (*

    Function application E1 E2.

    *)
  6. | Pexp_let of Ast_types.rec_flag * value_binding list * expression
    (*

    Let expressions: let P1 = E1 and ... and Pn = En in E

    *)
  7. | Pexp_forall of string list * expression
    (*

    Explicit "forall" quantifier forall 'a1 ... 'an -> E. Invariant n >= 1.

    *)
  8. | Pexp_exists of string list * expression
    (*

    Explicit "exists" quantifier exists 'a1 ... 'an -> E. Invariant n >= 1.

    *)
  9. | Pexp_constraint of expression * core_type
    (*

    An expression "constraint" or typing annotation (E : T).

    *)
  10. | Pexp_construct of string * expression option
    (*

    An applied algebraic data type constructor C <E>.

    *)
  11. | Pexp_record of (string * expression) list
    (*

    {l1 = E1; ...; ln = En}

    *)
  12. | Pexp_field of expression * string
    (*

    E.l

    *)
  13. | Pexp_tuple of expression list
    (*

    Tuples (E1, ..., En). Invariant: n >= 2.

    *)
  14. | Pexp_match of expression * case list
    (*

    Match (or "case") expressions match E with (P1 -> E1 | ... | Pn -> En).

    *)
  15. | Pexp_ifthenelse of expression * expression * expression
    (*

    If (or ternary) expressions if E then E1 else E2.

    *)
  16. | Pexp_try of expression * case list
    (*

    try E0 with P1 -> E1 | ... | Pn -> En

    *)
  17. | Pexp_sequence of expression * expression
    (*

    E1; E2

    *)
  18. | Pexp_while of expression * expression
    (*

    while E1 do E2 done

    *)
  19. | Pexp_for of pattern * expression * expression * Ast_types.direction_flag * expression
    (*

    for i = E1 to E2 do E3 done for i = E2 downto E2 do E3 done

    *)
  20. | Pexp_variant of string * expression option
    (*

    `A <E>

    *)
and value_binding = {
