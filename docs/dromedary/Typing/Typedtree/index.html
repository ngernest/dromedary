type expression = {
  1. exp_desc : expression_desc;
  2. exp_type : Types.type_expr;
}
and expression_desc =
  1. | Texp_var of string instance
    (*

    x.

    *)
  2. | Texp_prim of Parsing.Ast_types.primitive
    (*

    Primitive operations %prim p. e.g. %prim +, %prim -, etc.

    *)
  3. | Texp_const of Parsing.Ast_types.constant
    (*

    Constants c. e.g. 1, true, ().

    *)
  4. | Texp_fun of pattern * expression
    (*

    The function (or lambda) abstraction fun P -> E. Note that:

    • let x P1 ... Pn = E in ... is encoding using Texp_let ("x", fun P1 ... Pn -> E, ...).
    *)
  5. | Texp_app of expression * expression
    (*

    Function application E1 E2.

    *)
  6. | Texp_let of value_binding list * expression
    (*

    Let expressions

    *)
  7. | Texp_let_rec of rec_value_binding list * expression
    (*

    Let rec expressions

    *)
  8. | Texp_construct of Types.constructor_description * expression option
    (*

    An applied algebraic data type constructor C <E>.

    *)
  9. | Texp_record of (Types.label_description * expression) list
    (*

    {l1 = E1; ...; ln = En}

    *)
  10. | Texp_field of expression * Types.label_description
    (*

    E.l

    *)
  11. | Texp_tuple of expression list
    (*

    Tuples (E1, ..., En). Invariant: n >= 2.

    *)
  12. | Texp_match of expression * Types.type_expr * case list
    (*

    Match (or "case") expressions match E with (P1 -> E1 | ... | Pn -> En).

    *)
  13. | Texp_ifthenelse of expression * expression * expression
    (*

    If (or ternary) expressions if E then E1 else E2.

    *)
  14. | Texp_try of expression * case list
    (*

    try E0 with P1 -> E1 | ... | Pn -> En

    *)
  15. | Texp_sequence of expression * expression
    (*

    E1; E2

    *)
  16. | Texp_while of expression * expression
    (*

    while E1 do E2 done

    *)
  17. | Texp_for of string * expression * expression * Parsing.Ast_types.direction_flag * expression
    (*

    for i = E1 to E2 do E3 done for i = E2 downto E2 do E3 done

    *)
  18. | Texp_variant of Types.variant_description * expression option
    (*

    Polymorphic variant `A <E>.

    *)
